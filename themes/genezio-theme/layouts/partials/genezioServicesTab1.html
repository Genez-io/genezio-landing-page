genezio server

import { GenezioDeploy, GnzContext, GnzAuth } from '@genezio/types';

@GenezioDeploy()
export class UserInfoService {

  @GnzAuth
  async getUserInfo(context: GnzContext): Promise {
    const userInfo = await UserModel.getUserInfo(context.user.id)

    return userInfo
  }
}
