type Person = {
  firstName: string;
  lastName: string;
}

@GenezioDeploy()
class HelloWorldService {
  hello(name: string): string {
    return `Hello ${name}!`;
  }

  bye(person: Person): string {
    return `Bye, ${person.firstName} ${person.lastName}!`;
  }
}
client

import { HelloworldService, Person } from '@genezio-sdk/helloworld-app';

const hi = await HelloworldService.hello('John');

const bye = await HelloworldService.hello(person);

We parse the backend code you write and store the AST in a database. From there we generate client side stubs in any language. It's like tRPC but for any backend language to any frontend language. Now supporting Javasript/Typescript, with support for DART/Flutter, Kotlin, Swift, GO, Python and others coming soon.

Out of the box user auth. Currently supported methods are: email/password, google and github auth. Others are coming soon. What we do is generated the code for you as part of your project.

client

  // rest of the code

  const handleSignupSubmit = async (e) => {
    // Use the auto generated EmailAuthService to register the user
    await EmailAuthService.register(user.email, user.password, user.username)
  }

  const getUserInfo = async (e) => {
    // Use the auto generated EmailAuthService to register the user

@GenezioDeploy()
export class ProductService {
  async saveProductReview(context: GnzContext, review: ProductReview): Promise {
    const storage: GStorage = getStorage('productReviewImages');
      
    for (const image of review.images) {
      await storage.writeFile(image.name, image.content)
    }
  }
}
